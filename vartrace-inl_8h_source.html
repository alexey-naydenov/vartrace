54 VAR_TRACE_TEMPLATE
55 void VarTrace<LL, CP, LP, AP>::Initialize() {
56  Lock guard(*this);
57  // check for double initialization
58  if (is_initialized_) {return;}
59  // check block count size
60  if (block_count_ < kMinBlockCount) {return;}
61  // try to allocate storage
62  data_ = this->Allocate(block_count_*block_length_);
63  if (data_) {
64  is_initialized_ = true;
65  can_log_ = true;
66  // init blocks description variables
67  index_mask_ = (block_count_*block_length_) - 1;
68  block_end_indices_.reset(new int[block_count_]);
69  block_end_indices_[0] = 0; // start position of the cursor
70  for (unsigned i = 1; i < block_count_; ++i) {
71  block_end_indices_[i] = -1;
72  }
73  }
74 }

179 VAR_TRACE_TEMPLATE template <typename T>
180 void VarTrace<LL, CP, LP, AP>::DoLog(MessageIdType message_id, const T *value,
181  const SizeofCopyTag &copy_tag,
182  unsigned data_id, unsigned object_size) {
183  assert(current_index_ < block_count_*block_length_);
184  Lock guard(*this);
185  if (!can_log_) {return;}
186  CreateHeader(message_id, data_id, object_size);
187  // check if data fits in space left in trace
188  if ((block_count_*block_length_ - current_index_)
189  *sizeof(AlignmentType) > object_size) {
190  // copy using one function call
191  std::memcpy(&(data_[current_index_]), value, object_size);
192  // increment index
193  current_index_ += RoundSize(object_size);
194  current_index_ = current_index_ & index_mask_;
195  } else {
196  int copied_size = 0;
197  // copy till the end of the trace
198  int size_to_copy =
199  (block_count_*block_length_ - current_index_)
200  *sizeof(AlignmentType);
201  std::memcpy(&(data_[current_index_]), value, size_to_copy);
202  copied_size = size_to_copy;
203  // copy rest of data to the begging of the trace buffer
204  size_to_copy = object_size - copied_size;
205  std::memcpy(&(data_[0]), reinterpret_cast<const uint8_t *>(value)
206  + copied_size, size_to_copy);
207  current_index_ = RoundSize(size_to_copy);
208  }
209  current_block_ = current_index_ >> log2_block_length_;
210  block_end_indices_[current_block_] = current_index_;
211 }

226 VAR_TRACE_TEMPLATE
227 unsigned VarTrace<LL, CP, LP, AP>::DumpInto(void *buffer, unsigned size) {
228  Lock guard(*this);
229  if (!can_log_) {return 0;}
230  // start copying from the end of the next block
231  int copy_from = block_end_indices_[NextBlock(current_block_)];
232  // if end index of the next block is -1 then the trace was not
233  // filled even once, copy from index 0
234  if (copy_from < 0) { copy_from = 0; }
235  int copy_to = block_end_indices_[current_block_];
236  // size of data copied in bytes
237  int copied_size = 0;
238  // check if block being copied wraps around
239  if (copy_from > copy_to) { // wrapping
240  // check the part before end of trace fits in buffer
241  if ((block_length_*block_count_ - copy_from)
242  *sizeof(AlignmentType) <= size) { // copy all from copy_from till end
243  copied_size = (block_length_*block_count_ - copy_from)
244  *sizeof(AlignmentType);
245  std::memcpy(buffer, &(data_[copy_from]), copied_size);
246  // try to c opy from index 0 till copy_to
247  int leftover_to_copy = 0;
248  if (copy_to*sizeof(AlignmentType) <= size - copied_size) {
249  // everything fits
250  leftover_to_copy = copy_to*sizeof(AlignmentType);
251  } else { // only some part form 0 fits
252  leftover_to_copy = size - copied_size;
253  }
254  std::memcpy(static_cast<uint8_t *>(buffer) + copied_size,
255  &(data_[0]), leftover_to_copy);
256  copied_size += leftover_to_copy;
257  } else { // copy as much as possible from copy_from till end
258  copied_size = size;
259  std::memcpy(buffer, &(data_[copy_from]), copied_size);
260  }
261  } else { // no wrapping
262  // copy at most size bytes
263  if ((copy_to - copy_from)*sizeof(AlignmentType) <= size) {
264  copied_size = (copy_to - copy_from)*sizeof(AlignmentType);
265  } else {
266  copied_size = size;
267  }
268  std::memcpy(buffer, &(data_[copy_from]), copied_size);
269  }
270  return copied_size;
271 }

299  Lock guard(*this);

311 VAR_TRACE_TEMPLATE
312 void VarTrace<LL, CP, LP, AP>::SubtraceDestruction(
313  unsigned subtrace_current_index) {
314  Lock guard(*this);
315  // copy updated indices into parent logger
316  current_index_ = subtrace_current_index;
317  current_block_ = current_index_ >> log2_block_length_;
318  // allow logging
319  can_log_ = true;
320  // calculate written size
321  unsigned written_length = 0;
322  if (current_index_ < subtrace_start_index_) {
323  // trace buffer was wrapped around
324  written_length = (block_count_*block_length_
325  - subtrace_start_index_) + current_index_;
326  } else {
327  // no wrapping happenned
328  written_length = current_index_ - subtrace_start_index_;
329  }
330  // update size of the message that contains subtrace
331  data_[PreviousIndex(subtrace_start_index_)] |=
332  written_length*sizeof(AlignmentType);
333 }
